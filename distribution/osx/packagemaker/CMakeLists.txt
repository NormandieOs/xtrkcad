# Setup creation of an installer with PackageMaker
SET(CPACK_GENERATOR "PackageMaker")
SET(CPACK_PACKAGE_NAME "XTrkCAD")
SET(CPACK_PACKAGE_FILE_NAME "xtrkcad-setup-${XTRKCAD_VERSION}-${CMAKE_SYSTEM_PROCESSOR}")

SET(CPACK_RESOURCE_FILE_WELCOME "${CMAKE_CURRENT_SOURCE_DIR}/welcome.html")
SET(CPACK_RESOURCE_FILE_README "${CMAKE_CURRENT_SOURCE_DIR}/readme.html")

CONFIGURE_FILE(${XTrkCAD_SOURCE_DIR}/app/COPYING ${CMAKE_CURRENT_BINARY_DIR}/COPYING.txt COPYONLY)
SET(CPACK_RESOURCE_FILE_LICENSE "${CMAKE_CURRENT_BINARY_DIR}/COPYING.txt")

INCLUDE(CPack)
